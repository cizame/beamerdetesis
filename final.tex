\begin{frame}

\begin{alertblock}<+->{Gráfica}
Una gráfica $\Gamma$ es un par ordenado de conjuntos finitos,
  $\Gamma=(V(\Gamma),E(\Gamma))$, donde $V(\Gamma)$ es
  llamado vértices\index{Vértice} de $\Gamma$ y $E(\Gamma)$ aristas de
  $\Gamma$, tales que:
  \begin{equation*}
    E(\Gamma) \subseteq
    V(\Gamma)^{(2)}:=\{e\subseteq V(\Gamma) \mid |e|=2\}
  \end{equation*}
\end{alertblock}

\begin{columns}
\column{.5\textwidth}
\only<2-4>{
\begin{itemize}
\item<2-> Sin aristas dirigidas
\item<3-> Sin lazos
\item<4-> Sin aristas multiples
\end{itemize}
}

\only<1>{
\begin{alertblock}{Gráfica Bipartita}
Sea $B$ una gráfica. Si existen subconjuntos no vacíos $X$ y $Y$ de
  $V(B)$ tales que:
  \begin{enumerate}
  \item $V(B)=X\cup Y$ y $X\cap Y=\emptyset$.
  \item Si $x\sim y$ entonces $x\in X$ y $y\in Y$ o viceversa.
  \end{enumerate}
  entonces a $B$ se le conoce como gráfica bipartita. A los conjuntos
  $X$ y $Y$ se les conoce como biparticion de $B$.
\end{alertblock}

\only<2>{
\begin{alertblock}{Gráfica bipartita completa: $K_{m,n}$}
Sean $X$ y $Y$ conjuntos tales que forman una biparticion de una
gráfica bipartita $B$. Si dado $x\in V(B)$ para todo $y\in Y$ existe
arista entre $x$ y $y$, se dice que $B$ es una gráfica bipartita completa.

\begin{alertblock}{Gráfica $k$-regular}
Una gráfica $G$ es $k$-regular si todos sus vértices son adyacentes a
exactamente $k$ elementos.
\end{alertblock}

\only<2,3>{
\begin{alertblock}{Gráfica completa: $K_n$}
  Una gráfica $G$ es completa si para todo par de vértices $x,y\in
  V(G)$ existe arista entre ellos. Si $|V(G)|=n$ entonces $G$ es
  denotada por $K_n$.
\end{alertblock}}
}

\only<3>{
\begin{alertblock}{Ciclo de longitud $n$: $C_n$}
Un ciclo de longitud $n$ es una sucesión
  de vértices $v_1,...,v_n$ todos distintos entre sí, tales que
  $v_i$ es adyacente a $v_{i+1}$ con $i=1...n-1$ y $v_1$ es adyacente
  a $v_n$.
\end{alertblock}
}}
\end{columns}

\begin{frame}{Gráfica $k$-regular}

\only<1-3>{
\begin{columns}
\column{.4\textwidth}
\begin{alertblock}{Grado de un vértice}
El grado de un vértice es igual al número de aristas que lo contienen.
\end{alertblock}
\column{.4\textwidth}
\begin{alertblock}{Orden de una gráfica}
El orden de una gráfica es igual al número de vertices que tiene.
\end{alertblock}
\end{columns}}
\only<2-3>{
\begin{alertblock}{Teorema}
En toda gráfica $G$, la suma de los grados de
  sus vértices es igual al doble de la cantidad de aristas que tiene.
  \begin{equation*}
    \sum_{v\in V(G)}d_G(v)=2|E(G)|
  \end{equation*}
\end{alertblock}}
\only<3>{
\begin{alertblock}{Corolario}
Sea $G$ una gráfica $k$-regular con $k$ impar. Entonces el orden de
  $G$ es par.
\end{alertblock}}

\end{frame}

\begin{frame}{Cuello}

\begin{alertblock}{Cuello}
Sea $G$ una gráfica con ciclos. Se define el cuello de $G$ como:
  \begin{equation*}
    g(G):= \min\{ \text{longitud de $c$} \mid  \text{$c$ es un ciclo en $G$} \}
  \end{equation*}
%El cuello de una gráfica es la longitud de su ciclo más pequeño.

  Sean $\Gamma_1,\Gamma_2$ dos gráficas. Un isomorfismo de

  $\Gamma_1$ si y solo si $f(x)\thicksim f(y)$ en $\Gamma_2$.

\begin{frame}{Cota de Moore}

\begin{alertblock}{Teorema}
 Sea
  \begin{equation*}
    f_{0}(g)=
    \begin{cases}
      \sum^{m}_{i=1}2^i & \text{donde }m=\frac{g}{2}, \text{ si $g$ es par}\\
      1 + 3\sum^m_{i=0} 2^i & \text{donde }m=\frac{g-3}{2}, \text{ si $g$ es impar}
    \end{cases}
  \end{equation*}
  entonces, si $G$ es una $(3,g)$-gráfica, se tiene que $|G|\geq f_{0}(g)$.
\end{alertblock}

\begin{alertblock}{Gráfica bipartita clánica}
Sea $Cl_G:=\{H \mid H$ es un clan de una gráfica $G \}$. Se define
  la gráfica bipartita clánica $BK(G)$ de una gráfica $G$,
  como la gráfica con conjunto de vértices:
  $$V(BK(G)) := V(G) \dot{\cup} Cl_G$$ 
  y donde $x,y\in V(BK(G))$ son adyacentes si y solo si $x\in V(G)$ y
  $y$ es un clan de $G$, tales que $x\in y$ o viceversa.
\end{alertblock}

\begin{frame}
\begin{alertblock}{Gáfica de Cayley}
Sean $G$ un grupo y $T$ un subconjunto de $G$ tal que:
  \begin{enumerate}
  \item $e \notin T $
  \item Si $t\in T$ entonces $t^{-1}\in T$
  \end{enumerate}

  $g_1$ es adyacente a $g_2$ si y solo si $g_2^{-1}g_1 \in
  T$.
\end{alertblock}

\begin{alertblock}<+->{Teorema}
Sean $G$ un grupo y $H$ un subconjunto no vacío de $G$. Entonces las
  siguientes condiciones son equivalentes:
  \begin{enumerate}
  \item $H$ es un subgrupo de $G$
  \item \begin{enumerate}
    \item Para todo $x,y \in H$, $xy\in H$
    \item Para todo $x\in H$, $x^{-1}\in H$
    \end{enumerate}
  \item Para todo $x,y \in H$ se tiene $xy^{-1}\in H$
  \end{enumerate}\label{teonodemos}
\end{alertblock}

\begin{frame}

\begin{alertblock}<+->{Definición}
Un operador  lineal acotado $T:\mathcal{H}\to\mathcal{H}$ se dice
\textbf{hiperciclico} si para algún $x\in H$, $\text{Orb}(T,x)$ es
densa en $\mathcal{H}$.
\end{alertblock}

\begin{alertblock}<+->{Definición}
Un operador  lineal acotado $T:\mathcal{H}\to\mathcal{H}$ es
\textbf{hiperciclico en un subespacio M} si existe un vector
$x\in\mathcal{H}$ tal que $\text{Orb}(T,x)\bigcap M$ sea densa en
$M$.
\end{alertblock}

\end{frame}

\begin{frame}

\begin{prop}<+->[Kitai, 1982]
Si $T$ es invertible, $T$ es hiperciclico si y solo si $T^{-1}$ lo
es.
\end{prop}
\end{frame}

\begin{frame}

\begin{thm}[Salas, 1995]
Sea $T$ un desplazamiento bilateral con pesos hacia atrás en
$\ell^2(\Z)$ con su sucesión de pesos positivos y acotados
$\{a_n\}$. Entonces $T$ es hiperciclico si y solo si dado $\epsilon
>0 $y $q\in\Z $, existe una n arbitrariamente grande tal que para todo $\mid j\mid \leq
q$,


 $\Pi_{s=0}^{n-1}a_{s+j}>1/\epsilon $ y $\Pi_{s=1}^{n}a_{s-j}<\epsilon $

\end{thm}

\begin{thm}[Salas, 1995]

Sean $T_1$ y $T_2$ desplazamientos bilaterales con pesos con
sucesión de pesos $\{a_n\}$ y $\{b_n\}$. $T_1 \oplus T_2$ es
hiperciclico si y solo si  dado $\epsilon > 0$ y $q \in \N$ existe
una n suficientemente grande tal que para todo $\mid j\mid \leq q$

$$
\max \left\{ \prod_{k=0}^{n-1} a_{j+k}, \prod_{k=0}^{n-1} b_{j+k}
\right\} < \epsilon ,\hspace{10pt} \min \left\{ \prod_{k=1}^{n}
a_{j-k}, \prod_{k=1}^{n} b_{j-k} \right\} > 1/\epsilon
$$

\end{thm}

\end{frame}

,\frac{1}{2},\frac{1}{2},\frac{1}{2},2,2,\frac{1}{2},\textbf{1},2,\frac{1}{2},\frac{1}{2},2,2,2,\ldots\right\}$$
$$\{b_n\}_{n\in\Z}=\left\{\ldots

\end{example}
